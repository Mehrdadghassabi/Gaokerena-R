\chapter{جمع آوری دادگان}
\section{مقدمه}
برای این منظور، اقدام به جمع‌آوری داده‌ها صورت گرفت. داده‌های جمع‌آوری‌شده شامل یک پیکره پزشکی، یک مجموعه داده پرسش و پاسخ فرم آزاد
\footnote{free form question answering} 
و یک مجموعه داده پرسش و پاسخ چهار گزینه‌ای بودند. شیوه‌های جمع‌آوری این داده‌ها شامل خزش
\footnote{crawl}
، ترجمه انسانی، ترجمه ماشینی و تبدیل عکس به متن بودند که در ادامه به تفصیل به هر یک از این روش‌ها پرداخته خواهد شد؛ در این راستا، تلاش‌هایی برای استفاده از منابع مختلف و بهره‌گیری از تکنیک‌های نوین صورت گرفت تا داده‌هایی با کیفیت بالا و متنوع گردآوری شوند. این داده‌ها امکان انجام تحقیقات دقیق‌تر و دستیابی به نتایج قابل اعتماد را فراهم کردند.



\section{معرفی پیکره پزشکی فارسی}
عدم وجود یک پیکره پزشکی اختصاصی به زبان فارسی، چالشی قابل توجه برای پژوهشگران و توسعه‌دهندگانی ایجاد می‌کند که هدفشان توسعه مدل های پزشکی در زبان فارسی است. بدون داده‌های متنی باکیفیت و تخصصی که برای آموزش مدل‌های هوش مصنوعی ضروری است، این تلاش‌ها ممکن است با موانع روبه‌رو شوند و در نهایت بر توسعه فناوری‌ها و راه‌حل‌های پیشرفته پزشکی مناسب برای جمعیت فارسی‌زبان تاثیر بگذارند. برای حل این مشکل، ما یک پیکره
\footnote{\lr{corpus}}
جامع شامل تقریبا نود میلیون توکن و حدود صد هزار مقاله گردآوری کرده‌ایم.
\footnote{
	برای بازدید از این پیکره میتوانید به آدرس
	\lr{huggingface.co/datasets/gaokerena/medical\_corpus}

\cite{b55}
مجموعه‌ای از متون پزشکی را که به چهار زبان (انگلیسی، فرانسوی، اسپانیایی و ایتالیایی) اختصاص داشت، گردآوری کردند که می‌توان آن را همانطور که در جدول

\section{معرفی مجموعه داده 
	\lr{MF3QA}
}
گردآوری یک مجموعه داده واقعی از پرسش‌ وپاسخ‌های پزشک و بیمار(پرسش و پاسخ فرم آزاد)
اهمیت بسیاری در ارتقا توانایی‌های مدل‌های زبانی در حوزه بهداشت و درمان دارد. چنین مجموعه داده‌ای به مدل‌های زبانی امکان می‌دهد تا اطلاعات ارزشمندی را که از تعاملات واقعی میان ارائه‌دهندگان خدمات بهداشتی و بیماران به دست می‌آید، بیاموزند. با تحلیل این تعاملات واقعی، مدل‌های زبانی می‌توانند به درک جزئیات اصطلاحات پزشکی، نگرانی‌های بیماران، و زمینه پیرامون سوالات بهداشتی دست یابند. علاوه بر این، این مجموعه داده مدل‌ها را قادر می‌سازد نه تنها محتوای دقیق پاسخ‌ها، بلکه ساختار و لحن مناسب برای پاسخ‌دهی به سوالات را نیز یاد بگیرند. این فرآیند دوگانه یادگیری از اهمیت بالایی برخوردار است، زیرا به مدل امکان می‌دهد پاسخ‌هایی دقیق، همدلانه و متناسب با زمینه ارائه دهد و در نهایت ارتباط و پشتیبانی از بیماران در محیط‌های پزشکی را بهبود بخشد.

در این راستا ما مجموعه داده
\lr{MF3QA}
که شامل بیست هزار جفت پرسش و پاسخ پزشکی در مجموعه تمرین
\footnote{\lr{train set}}
،دو هزار جفت پرسش و پاسخ پزشکی در مجموعه توسعه
\footnote{\lr{development set}}
و دو هزار جفت پرسش و پاسخ پزشکی در مجموعه آزمایش
\footnote{\lr{test set}}
است را معرفی کردیم.همچنین در این زمینه، یانگ لیو در مقاله مروری
\footnote{\lr{survey}}
خود 
\cite{b56}
به چندین مجموعه داده واقعی پرسش‌ وپاسخ پزشک وبیمار اشاره کرده است، مقایسه‌ای میان این مجموعه‌ دادگان و مجموعه داده ما در جدول

\subsection{
	فیلتر کردن رکورد های مجموعه داده
	\lr{MF3QA}
}
در پایان‌نامه حاضر، بیش از صد و هشتاد هزار جفت پرسش‌وپاسخ از تالارهای گفت‌وگوی پزشکی فارسی گردآوری شده است. این جفت‌های پرسش‌وپاسخ، چه به صورت دستی
\footnote{
	فرآیند فیلتر کردن دستی توسط خانم زهرا کاظمی و آقای میلاد توکلی، از دانشجویان کارشناسی مهندسی کامپیوتر، انجام شده است.
}
و چه به صورت خودکار، مورد بررسی قرار گرفته و جفت‌هایی که حاوی اطلاعات مفید نبودند، حذف شده‌اند.
\footnote{
	برای بازدید از مجموعه داده 
	\lr{MF3QA}
	به آدرس
	\lr{huggingface.co/datasets/gaokerena/MF3QA}
	و برای بازدید از صد و هشتاد هزار جفت پرسش‌وپاسخ خزش شده به آدرس
	\lr{huggingface.co/datasets/gaokerena/MF3QA\_uncleaned}
	مراجعه کنید.
}

این رویکرد مشابه کاری است که یونشیانگ لی و همکارانش برای توسعه مدل زبانی
\lr{Chat Doctor}
انجام داده‌اند.
\cite{b40}
آنها نیز داده‌ها را از تالارهای گفت‌وگوی پزشکی انگلیسی استخراج کرده و نیمی از جفت‌های پرسش‌وپاسخ را بر اساس طول پاسخ‌ها کنار گذاشته‌اند
\footnote{
	فیلتر کردن آنها صرفا بر اساس طول پاسخ بوده ولی همانطور که پیشتر اشاره شد ما برای فیلتر کردن از روش های دستی نیز استفاده کرده ایم.
}
، 
چراکه پاسخ‌های کوتاه‌تر معمولا حاوی اطلاعات مفیدی نیستند. با این حال، ما با چالش بزرگ‌تری مواجه بودیم؛ پزشکان فارسی‌زبان معمولا پاسخ‌های بسیار کوتاه‌تری نسبت به همتایان انگلیسی خود ارائه می‌دهند. این امر ما را مجبور کرد تا بیش از هشتاد درصد از رکوردهای پرسش‌وپاسخ خود را برای تضمین کیفیت کنار بگذاریم.

\subsubsection{
	خزش از تالار گفتگو دکترهست
}
خزش از تالار گفتگوی “دکترهست”، که اصلی‌ترین منبع مجموعه داده
\lr{MF3QA}
است، با چالش خاصی همراه بود. این تالار گفتگو تمام رکوردهای تعامل پزشک و بیمار خود را به صورت مستقیم در سایت ارائه نمی‌دهد و فقط به دو هزار رکورد آخر دسترسی می‌دهد. علاوه بر این، هر رکورد به صد رکورد مرتبط دیگر پیوند داده شده است.

برای حل این چالش، از الگوریتم
\ref{alg:BFS}
استفاده شد. در این روش، داده‌های تالار گفتگو به صورت یک گراف در نظر گرفته شده و با استفاده از جستجوی عرض-اول 
\footnote{\lr{breadth first search}}
توانستیم حدود صد و بیست هزار رکورد از مجموع دویست هزار رکورد موجود در این تالار گفتگو را استخراج کنیم. این فرایند حدود دو هفته طول کشید.

\begin{algorithm}[ht]
	\onehalfspacing
	\caption{
		الگوریتم جستجو اول عرض برای استخراج رکورد های پرسش و پاسخ پزشکی
	} \label{alg:BFS}
	\begin{algorithmic}[1]
		\REQUIRE گره های دارای دسترسی در تالار گفتگو (برگ ها)		
		\\
		\ENSURE مجموعه‌ای از گره‌های بازدید شده
		\STATE یک پشته خالی $S$ ایجاد کن
		\STATE یک مجموعه خالی $Visited$ ایجاد کن
		\STATE گره مبدأ $v$ را به پشته $S$ اضافه کن
		\WHILE{پشته $S$ خالی نیست}
		\STATE یک گره $u$ را از پشته $S$ بردار
		\IF{گره $u$ بازدید نشده است}
		\STATE گره $u$ را به مجموعه $Visited$ اضافه کن
		\FOR{هر همسایه $n$ از گره $u$}
		\IF{گره $n$ بازدید نشده است}
		\STATE گره $n$ را به پشته $S$ اضافه کن
		\ENDIF
		\ENDFOR
		\ENDIF
		\ENDWHILE
		\RETURN نود های بازدید شده
		
	\end{algorithmic}
\end{algorithm}

\section{معرفی مجموعه داده پرسش های چند گزینه ای}
جمع‌آوری مجموعه پرسش‌های چند گزینه‌ای نیز مانند مجموعه فرم آزاد دارای اهمیت است. این موضوع از آن جهت حائز اهمیت است که برای تمرین دادن یک مدل زبانی، در برخی شیوه‌های معرفی شده مانند شیوه تمرین مدل
\lr{medsss}
\cite{b32}
یا شیوه پیشنهادی در همین مقاله که در فصل ششم به آن پرداخته خواهد شد، ما نیازمند پرسش‌هایی هستیم که وقتی مدل زبانی به آن‌ها پاسخ می‌دهد، به سادگی مشخص باشد که پاسخ نهایی صحیح است یا خیر! این ویژگی در پرسش‌های فرم آزاد وجود ندارد و برای تشخیص میزان درستی آن نیازمند یک داور
\footnote{\lr{judge}}
هستیم. بنابراین، برای سنجش مدل زبانی نیز از همین نظر دارای اهمیت است! زیرا در پرسش‌های فرم آزاد برای سنجیدن پاسخ باز نیازمند یک داور هستیم، ولی پرسش‌های چند گزینه‌ای دارای یک گزینه مشخص به عنوان پاسخ صحیح هستند.

به همین منظور ما اقدام به گردآوری این پرسش و پاسخ‌ها از منابع گوناگون کردیم. این منابع شامل ترجمه انسانی پرسش‌های انگلیسی، استفاده از کنکور علوم پایه پزشکی ایران و در نهایت استفاده از ترجمه ماشینی پرسش‌های انگلیسی بودند. در مجموع حدود هزار و سیصد پرسش برای تمرین، حدود هزار و هفتصد پرسش برای توسعه و حدود هجده هزار پرسش و پاسخ برای مجموعه تمرین حاصل شد. در نظر داشته باشید که ترجمه ماشینی تنها برای مجموعه تمرین استفاده شده است تا از خطاهای آن در آزمایش و توسعه جلوگیری شود.

\subsection{ترجمه قسمت پزشکی مجموعه داده
	\lr{MMLU}
}
مجموعه داده 
\lr{MMLU}
\cite{b14}
\footnote{\lr{massive multitask language understanding}}
یکی از معتبرترین مجموعه داده‌ها برای ارزیابی توانایی مدل‌های زبانی در درک و پاسخ‌دهی به سوالات چندوظیفه‌ای است. این مجموعه شامل سوالاتی در موضوعات مختلف از جمله علوم پزشکی، مهندسی، علوم انسانی و دیگر حوزه‌ها است که به‌صورت چندگزینه‌ای طراحی شده‌اند.

برای ارزیابی دانش پزشکی مدل های زبانی توسعه‌یافته، بخش پزشکی این مجموعه داده را به زبان فارسی ترجمه شده است.
\footnote{
ترجمه توسط آقای امیرحسین پورسینا دانشجوی پزشکی انجام شده است.
}
 هدف از این کار، تطبیق داده‌های ارزیابی با زبان مورد استفاده در مدل و بررسی توانایی مدل در پاسخ‌دهی دقیق به سوالات تخصصی پزشکی در زبان فارسی بوده است.
\footnote{
برای بازدید از ترجمه این مجموعه داده به آدرس
\lr{huggingface.co/datasets/gaokerena/FA\_MED\_MMLU}
مراجعه کنید.
 }
\subsection{گردآوری سوالات کنکور علوم پایه پزشکی ایران
}
آزمون علوم پایه پزشکی یک آزمون سراسری در ایران است که دانشجویان پزشکی موظف هستند پس از گذراندن دروس علوم پایه معمولا در پنج ترم در آن شرکت کنند. این آزمون به منظور سنجش میزان آموخته‌های دانشجویان از دروس علوم پایه و آمادگی آنها برای ورود به مراحل بالینی برگزار می‌شود. در صورتی که دانشجو پس از سه مرتبه در این آزمون قبول نشود
\footnote{نمره قبولی در این آزمون در سالهای مختلف متفاوت است و معمولا حدود سی و شش درصد میباشد}
از ادامه تحصیل در رشته پزشکی محروم میشود.

سوالات این آزمون ابتدا از سایت سازمان سنجش به صورت
\lr{pdf}
دانلود شده سپس در یک خط لوله
\footnote{\lr{pipeline}}
تبدیل تصویر به متن که شامل انسان و ماشین بوده است این سوالات به صورت فایل اکسل در آمده است؛
\footnote{برای این کار از کتابخانه
\lr{fitz}
پایتون استفاده شده است.
}
لازم به ذکر است که از سوالات کنکور علوم پزشکی پایه شهریور 
\lr{1402}
در مجموعه آزمایش و از سوالات بقیه سال ها در مجموعه توسعه استفاده شده است.
\footnote{
	برای بازدید از این مجموعه داده به آدرس

	مراجعه کنید.

\subsection{ترجمه ماشینی مجموعه داده
	\lr{MedMCQA}
}
برای پیاده‌سازی روش پیشنهادی خود در فاز دوم که در فصل ششم به آن به طور مفصل پرداخته خواهد شد، نیاز به تعداد بالایی پرسش و پاسخ پزشکی چند گزینه ای داشتیم؛ از آنجایی که سوال های کنکور علوم پزشکی پایه کفاف تعداد بالای پرسش های مورد نیاز برای تمرین را نمیدهد و از آنجا که در زمان انجام پژوهش، چنین مجموعه‌داده‌ای در دسترس نبود، تصمیم گرفتیم از مدل زبانی بزرگ 
\lr{DeepSeek-V3}
\cite{b63}
که از نظر هزینه و کارایی مقرون‌به‌صرفه است، برای ترجمه‌ی بخشی از مجموعه‌داده‌ی
\lr{MedMCQA}
\cite{b64}
از زبان انگلیسی به فارسی استفاده کنیم.
\footnote{
ما تنها برای مجموعه تمرین از پرسش هایی که توسط ترجمه ماشینی تولید شده اند استفاده کرده ایم زیرا مجموعه تمرین حساسیت کمتری نسبت به مجموعه های آزمایش و توسعه در صحت و درستی پرسش ها دارد.
}

به‌منظور حفظ تنوع موضوعی، پرسش‌ها به‌صورت تصادفی از مجموعه‌داده‌ی
\lr{MedMCQA}
انتخاب شدند تا دامنه‌ی وسیعی از مباحث علوم پزشکی (از جمله فیزیولوژی، داروشناسی، پاتولوژی، و سایر حوزه‌ها) در مجموعه‌ی نهایی پوشش داده شود.
\footnote{
	برای بازدید از این مجموعه داده به آدرس
	\lr{github.com/Mehrdadghassabi/Gaokerena-R/tree/main/dataset/medmcqa\_translation\_filterd}
	مراجعه کنید.
}

برای اطمینان از کیفیت ترجمه‌ها، از دو داور زبانی مستقل به نام‌های 
\lr{grok-3-mini}
\cite{b65}
و 
\lr{gpt-4.1-mini}
\cite{b66}
استفاده کردیم. هر یک از این داوران موظف بودند که ترجمه‌ی هر پرسش و گزینه‌های آن را از نظر دقت معنایی، روانی زبانی و حفظ اصطلاحات تخصصی ارزیابی کنند. ترجمه تنها در صورتی به عنوان «تاییدشده» در نظر گرفته می‌شد که هر دو داور به آن امتیاز کامل پنج از پنج اختصاص دهند.در پایان این فرآیند چندمرحله‌ای، موفق شدیم حدود هجده هزار پرسش چندگزینه‌ای پزشکی به زبان فارسی را که از نظر کیفیت ترجمه تایید شده بودند، گردآوری کنیم.

\section{
مجموعه دادگان معرفی شده پس از گردآوری
}

پس از اتمام فاز جمع‌آوری داده‌ها و انتشار آن‌ها به صورت عمومی، چند مقاله مهم در حوزه جمع‌آوری مجموعه‌داده‌های پزشکی منتشر شد. در این قسمت به معرفی این مجموعه داده ها خواهیم پرداخت هرچند که از این دادگان در این پایان نامه استفاده نشده است.
\subsection{
مجموعه داده
\lr{PerMedCQA}
}
مجموعه‌داده \lr{PerMedCQA} \cite{b67} از نظر ساختار و نوع پرسش‌ها شباهت زیادی به مجموعه‌داده
\lr{MF3QA}
دارد. مانند
\lr{MF3QA}
، این مجموعه داده نیز یک مجموعه پرسش و پاسخ فرم آزاد پزشکی است که از وب‌سایت‌های مختلف جمع‌آوری شده و سپس توسط انسان و ماشین فیلتر شده است.
\footnote{
	برای بازدید از این مجموعه داده به آدرس
	\lr{huggingface.co/datasets/NaghmehAI/PerMedCQA}
	مراجعه کنید.
}
نویسنده مقاله 
\lr{PerMedCQA}
ادعا کرده که این مجموعه‌داده نخستین در نوع خود است، اما این ادعا نادرست است، زیرا مجموعه‌داده
\lr{MF3QA}
پیش از 
\lr{PerMedCQA}
معرفی شده بود.
\footnote{
آیدی آرکایو دو مقاله نیز گواه برا این موضوع است، آیدی مقاله ای که در آن
\lr{MF3QA}
معرفی شده است
\lr{2505.16000}
میباشد و آیدی مقاله ای که در آن
\lr{PerMedCQA} 
معرفی شده
\lr{2505.18331}
است.
}

\subsection{
مجموعه داده
\lr{PersianMedQA}
}
مجموعه‌داده 
\lr{PersianMedQA}
\cite{b68}
شامل پرسش‌های چهارگزینه‌ای پزشکی به زبان فارسی است. این مجموعه می‌توانست منبع بسیار مناسبی برای پژوهش حاضر باشد؛ با این حال، تا زمان نگارش این مقاله (بهمن 
\lr{1404})، 
داده‌های آن به‌صورت عمومی در دسترس قرار نگرفته بود. از این‌رو، ناچار به استفاده از ترجمه ماشینی مجموعه‌داده
\lr{MedMCQA}
شدیم که پیشتر در مورد آن صحبت شد.
\footnote{
	برای بازدید از این مجموعه داده به آدرس
	\lr{huggingface.co/datasets/MohammadJRanjbar/PersianMedQA}
	مراجعه کنید.
}

مجموعه داده

\footnote{
